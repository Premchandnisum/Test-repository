When /^I retrieve product price for item (\d+)$/ do |number|
  product_list = @ids_data_file['product_list']
  product_price_section = @ids_data_file['product_price_section']
  product_price = @ids_data_file['product_price']
  price_sale = @ids_data_file['price_sale']
  @item_dollar_price_as_string = "0"
  @item_dollar_price_as_float = 0.0  
  @has_sale = page.has_xpath?("//div[@id='#{product_list}']/div[#{number}]/div[@class='#{product_price_section}']/div[@class='#{product_price}']/span[@class='#{price_sale}']")
  if @has_sale == false
    item_price = page.find(:xpath, "//div[@id='#{product_list}']/div[#{number}]/div[@class='#{product_price_section}']/div[@class='#{product_price}']/span[1]").text   
  else
    item_price = page.find(:xpath, "//div[@id='#{product_list}']/div[#{number}]/div[@class='#{product_price_section}']/div[@class='#{product_price}']/span[@class='#{price_sale}']").text       
  end
  @item_dollar_price_as_string = item_price[(item_price.index('$')+1)..-1]
  @item_dollar_price_as_float = Float(@item_dollar_price_as_string)
  puts "Item dollar price: " 
  puts @item_dollar_price_as_float
end
